interface User {

      .map(part => part[0])
      .join('')
      .toUpperCase();

          <div style={{ width: '48px', height: '48px', borderRadius: '50%',

            {getInitials(name)}
          </div>

          <p style={{ margin: '0', color: '#6c757d' }}>{email}</p>

      </div>
    </div>

};
